Add a song to a playlist

Using the Playlist model, we will add a song to a playlist, given its URI.

The Javascript

window.onload = function() {
    var sp = getSpotifyApi(1);
    var models = sp.require('sp://import/scripts/api/models');

    // Create a track object from a URI
    var track = models.Track.fromURI('spotify:track:6fycP4n7KoYxnT169sLush');

    // Create a playlist object from a URI
    var playlist = models.Playlist.fromURI('spotify:user:magerleagues:playlist:3S0u0nTWj4P2bF5oq0DzLl');

    // Add the track
    playlist.add(track);

    // Verify the song was added to the playlist
    console.log(playlist);
}
